(function initBGM() {
  const audio = document.getElementById('bgm-audio');
  const btn   = document.getElementById('bgm-toggle');
  let playing = true;

  function tryPlay() {
    audio.play().catch(() => {});
  }

  tryPlay();

  function resumeOnInteraction() {
    if (playing && audio.paused) tryPlay();
    document.removeEventListener('click', resumeOnInteraction);
    document.removeEventListener('keydown', resumeOnInteraction);
  }
  document.addEventListener('click', resumeOnInteraction);
  document.addEventListener('keydown', resumeOnInteraction);

  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    playing = !playing;
    if (playing) {
      tryPlay();
      btn.setAttribute('aria-label', 'Tắt nhạc');
      btn.classList.remove('muted');
    } else {
      audio.pause();
      btn.setAttribute('aria-label', 'Bật nhạc');
      btn.classList.add('muted');
    }
  });
})();

(function initParticles() {
  const canvas = document.getElementById('particle-canvas');
  const ctx    = canvas.getContext('2d');
  let W, H, particles;

  function resize() {
    W = canvas.width  = window.innerWidth;
    H = canvas.height = window.innerHeight;
  }

  function createParticles() {
    particles = Array.from({ length: 55 }, () => ({
      x:    Math.random() * W,
      y:    Math.random() * H,
      r:    0.4 + Math.random() * 1.2,
      vx:   (Math.random() - 0.5) * 0.18,
      vy:   -0.08 - Math.random() * 0.18,
      a:    Math.random(),
      va:   (Math.random() - 0.5) * 0.004,
    }));
  }

  function draw() {
    ctx.clearRect(0, 0, W, H);
    for (const p of particles) {
      p.x  += p.vx;
      p.y  += p.vy;
      p.a  += p.va;
      if (p.a < 0) { p.a = 0; p.va = Math.abs(p.va); }
      if (p.a > 1) { p.a = 1; p.va = -Math.abs(p.va); }
      if (p.y < -10) { p.y = H + 10; p.x = Math.random() * W; }
      if (p.x < -10) { p.x = W + 10; }
      if (p.x > W + 10) { p.x = -10; }

      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
      ctx.fillStyle = `hsla(44, 65%, 60%, ${p.a * 0.55})`;
      ctx.fill();
    }
    requestAnimationFrame(draw);
  }

  resize();
  createParticles();
  draw();
  window.addEventListener('resize', () => { resize(); createParticles(); });

function _findGetAllKungfu() {
  const candidates = [window, window.parent, window.top];
  try { if (window.parent?.parent) candidates.push(window.parent.parent); } catch(e){}
  for (const w of candidates) {
    try { if (typeof w?.WX_GET_ALL_KUNGFU === 'function') return w.WX_GET_ALL_KUNGFU; } catch(e){}
  }
  return null;
}

function renderStats() {
  return STAT_KEYS.map(key => {
    // Đảm bảo không bao giờ bị NaN bằng cách dùng fallback
    const val = state.stats[key] ?? BASE_VAL;
    const pct = ((val - BASE_VAL) / TOTAL_PTS) * 100;
    return `
      <div class="stat-row" id="row-${key}">
        <div class="stat-name">${key.replace(/_/g, ' ')}</div>
        <div class="stat-bar-wrap">
          <div class="stat-bar" id="bar-${key}" style="width:${Math.max(pct,0)}%"></div>
        </div>
        <div class="stat-val" id="val-${key}">${val}</div>
        <div class="stat-btns">
          <button class="stat-btn" id="minus-${key}" onclick="adjustStat('${key}',-1)" aria-label="Giảm ${key}">−</button>
          <button class="stat-btn" id="plus-${key}"  onclick="adjustStat('${key}',+1)"  aria-label="Tăng ${key}">＋</button>
        </div>
      </div>`;
  }).join('');
}

function buildProgress() {
  const row = document.getElementById('progress-row');
  row.innerHTML = PAGES.map((_,i) => `<div class="prog-dot" id="dot-${i}" aria-hidden="true"></div>`).join('');
}

function updateProgress() {
  PAGES.forEach((_,i) => {
    const d = document.getElementById('dot-' + i);
    d.className = 'prog-dot' + (i === currentPage ? ' active' : i < currentPage ? ' done' : '');
  });
  document.getElementById('page-counter').textContent = `${currentPage+1} / ${PAGES.length}`;
  document.getElementById('btn-prev').disabled = currentPage === 0;

  const curId = PAGES[currentPage].id;
  const hideNext = curId === 'choose_path' || curId === 'transmigrate';
  const isLast = currentPage === PAGES.length - 1;
  document.getElementById('btn-next').style.display    = (isLast || hideNext) ? 'none' : '';
  document.getElementById('btn-confirm').style.display = isLast ? '' : 'none';
}

function buildPages() {
  const wrap = document.getElementById('page-wrap');
  wrap.innerHTML = '';
  PAGES.forEach((p, i) => {
    const div = document.createElement('div');
    div.className = 'page' + (i === 0 ? ' active' : '');
    div.id = 'page-' + p.id;
    div.setAttribute('role', 'region');
    div.setAttribute('aria-label', p.label);
    div.innerHTML = getPageHTML(p.id);
    wrap.appendChild(div);
  });
}

function navigate(dir) {
  if (PAGES[currentPage].id === 'player1') syncPlayer1();

  const oldPage = document.getElementById('page-' + PAGES[currentPage].id);
  oldPage.classList.remove('active');
  oldPage.classList.add('exit');

  const curId = PAGES[currentPage].id;

  if (dir === 1 && curId === 'choose_path') {
    if (state.path === 'transmigrate') {
      currentPage = PAGES.findIndex(p => p.id === 'transmigrate');
    } else {
      currentPage = PAGES.findIndex(p => p.id === 'player1');
    }
  } else if (dir === -1 && (curId === 'transmigrate' || curId === 'player1')) {
    currentPage = PAGES.findIndex(p => p.id === 'choose_path');
  } else {
    currentPage = Math.max(0, Math.min(PAGES.length - 1, currentPage + dir));
  }

  setTimeout(() => { oldPage.classList.remove('exit'); }, 600);

  const newPage = document.getElementById('page-' + PAGES[currentPage].id);
  void newPage.offsetWidth; // Force reflow
  newPage.classList.add('active');

  updateProgress();

  if (PAGES[currentPage].id === 'player2') {
    document.getElementById('stats-panel').innerHTML = renderStats();
    document.getElementById('pts-left').textContent  = state.pointsLeft;
    updateStatButtons();
  }
}

function adjustStat(key, delta) {
  if (delta > 0 && state.pointsLeft <= 0) return;
  if (delta < 0 && state.stats[key] <= BASE_VAL) return;
  state.stats[key]  += delta;
  state.pointsLeft  -= delta;

  document.getElementById('val-' + key).textContent = state.stats[key];

  const ptsEl = document.getElementById('pts-left');
  ptsEl.textContent = state.pointsLeft;
  ptsEl.classList.remove('flash');
  void ptsEl.offsetWidth;
  ptsEl.classList.add('flash');

  const pct = ((state.stats[key] - BASE_VAL) / TOTAL_PTS) * 100;
  document.getElementById('bar-' + key).style.width = Math.max(pct, 0) + '%';
  updateStatButtons();
}

function updateStatButtons() {
  STAT_KEYS.forEach(key => {
    const plus  = document.getElementById('plus-'  + key);
    const minus = document.getElementById('minus-' + key);
    if (plus)  plus.disabled  = state.pointsLeft <= 0;
    if (minus) minus.disabled = state.stats[key] <= BASE_VAL;
  });
}

function validate() {
  syncPlayer1();
  if (!state.gender)      { alert('Vui lòng chọn giới tính'); return false; }
  if (!state.appearance)  { alert('Vui lòng miêu tả ngoại hình'); return false; }
  if (!state.personality) { alert('Vui lòng miêu tả tính cách'); return false; }
  if (!state.openingId)   { alert('Vui lòng chọn mở đầu'); return false; }
  const oc = OPENING_CHOICES.find(c => c.id === state.openingId);
  if (oc && oc.editable && !state.openingCustom) { alert('Vui lòng miêu tả mở đầu tùy chỉnh'); return false; }
  if (!state.kungfu)      { alert('Vui lòng chọn võ học ban đầu'); return false; }

}

        _kungfuLibCache = lib;
        allKungfuRecord = Object.fromEntries(
          Object.values(lib).map(k => [k.Tên, k.Phân_loại || 'Thường'])
        );
      } else {
        try {
          const priorData = await Mvu.getMvuData({ type: 'message', message_id: 0 });
          const priorKungfu = priorData?.stat_data?.Võ_công;
          if (priorKungfu && Object.keys(priorKungfu).length > 5) {
            allKungfuRecord = priorKungfu;
          } else {
            throw new Error('Thư viện Võ công trong MVU không đầy đủ');
          }
        } catch(mvuErr) {
          allKungfuRecord = Object.fromEntries(KUNGFU_LIST.map(k => [k.id, 'Thường']));
        }
      }
    }
    _.set(data, 'stat_data.Võ_công', allKungfuRecord);

    await Mvu.replaceMvuData(data, { type: 'message', message_id: 0 });
    fillInput();
    showSuccess();
    await writeWorldbook();

  } catch(e) {
    console.error('[Khung Mở đầu] Lỗi:', e);
    btn.disabled = false;
    btn.innerHTML = `
      <svg style="width:16px;height:16px;stroke:currentColor;fill:none;stroke-width:1.5;flex-shrink:0;" viewBox="0 0 24 24">
        <path d="M5 12h14M12 5l7 7-7 7"/>
      </svg>
      Bước Vào Giang Hồ`;
    alert('Đã xảy ra lỗi: ' + e.message);
  }
}

async function writeWorldbook() {
  try {
    const parent = window.parent;
    if (!parent) return;

    const getNames     = parent.getCharWorldbookNames  || window.getCharWorldbookNames;
    const createEntries = parent.createWorldbookEntries || window.createWorldbookEntries;
    if (!getNames || !createEntries) return;

    const books = getNames('current');
    if (!books?.primary) return;

    const entryName = `${state.gender}·${state.kungfu}·Tùy chỉnh`;
    const content   = `Thông tin về {{user}}\nNgoại hình: ${state.appearance}\nTính cách: ${state.personality}`;

    await createEntries(books.primary, [{
      name:     entryName,
      strategy: { type: 'constant', keys: [] },
      position: { type: 'before_character_definition', order: 100 },
      content:  content,
      enabled:  true,
      probability: 100,
      recursion: { prevent_incoming: true, prevent_outgoing: true, delay_until: null },
      effect:   { sticky: null, cooldown: null, delay: null },
    }]);

    showWorldbookNotice(entryName);

  } catch(e) {
    console.warn('[Khung Mở đầu] Bỏ qua vì Lỗi viết Worldbook:', e.message);
  }
}

function showWorldbookNotice(entryName) {
  const overlay = document.getElementById('wb-notice-overlay');
  const text    = document.getElementById('wb-notice-text');
  if (text) text.textContent =
    `Đã tạo mục Worldbook「${entryName}」 (Đèn xanh · Luôn kích hoạt). ` +
    `Mục này sẽ không tự động xóa——Nếu cần bắt đầu chơi lại, vui lòng xóa thủ công mục này trong Worldbook trước khi tạo cuộc trò chuyện mới.`;
  if (overlay) {
    setTimeout(() => { overlay.style.display = 'flex'; }, 1200);
  }
}

document.getElementById('wb-notice-overlay').addEventListener('click', function(e) {
  if (e.target === this) this.style.display = 'none';
});

function fillInput() {
  const docs = [];
  try { if (window.top?.document)    docs.push(window.top.document);    } catch(e){}
  try { if (window.parent?.document) docs.push(window.parent.document); } catch(e){}
  docs.push(document);

  let textarea = null;
  for (const doc of docs) {
    try {
      textarea = doc.getElementById('send_textarea');
      if (textarea) break;
    } catch(e) {}
  }

  const oc = OPENING_CHOICES.find(c => c.id === state.openingId);
  const msg = oc && oc.editable
    ? `Dựa theo thiết lập mở đầu: ${state.openingCustom}, hãy bắt đầu đi.`
    : `Dựa theo thiết lập mở đầu, hãy bắt đầu kể chuyện đi.`;

  if (textarea) {
    textarea.value = msg;
    textarea.dispatchEvent(new Event('input',  { bubbles: true }));
    textarea.dispatchEvent(new Event('change', { bubbles: true }));
    textarea.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true }));
    textarea.focus();
  }
}

function showSuccess() {
  const overlay = document.getElementById('success-overlay');
  overlay.classList.add('show');
}
